# PostgreSQL 기본 command

```shell
# 모든 데이터베이스 표시
\l

# 특정 데이터베이스에 연결
\c database_name

# 현재 데이터베이스의 모든 테이블을 표시
\dt

# 특정 테이블을 세부정보 표시
\d table_name

# 특정 테이블의 모든 인덱스를 표시
\di table_name

# 모든 스키마 표시
\dn

# 현재 데이터베이스에 클러스터의 모든 역할(사용자) 표시
\du

# 현재 데이터베이스에 있는 개체의 액세스 권한을 표시
\dp

# 종료
\q

# 현재 사용자 및 데이터베이스를 표시
select current_user, current_database();

# 현재 데이터베이스 표시
\conninfo

# 현재 PostgreSQL 버전을 표시
select version();

# 데이터베이스 생성
create database <database>;
```
